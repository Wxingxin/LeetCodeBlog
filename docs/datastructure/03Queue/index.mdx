```js
class Queue {
  constructor() {
    this.items = [];
  }

  // 入队
  enqueue(element) {
    this.items.push(element);
  }

  // 出队
  dequeue() {
    return this.items.shift();
  }

  // 查看队头
  front() {
    return this.items[0];
  }

  // 判断是否为空
  isEmpty() {
    return this.items.length === 0;
  }

  // 队列长度
  size() {
    return this.items.length;
  }

  // 清空队列
  clear() {
    this.items = [];
  }
}

```
