# 方法 1

```js
var twoSum = function (nums, target) {
  //外层循环遍历每个元素
  for (let i = 0; i < nums.length; i++) {
    // 内层循环从 i+1开始，避免重复使用同一个元素
    for (let j = i + 1; j < nums.length; j++) {
      // 检查两数之和是否等于目标值
      if (nums[i] + nums[j] === target) {
        //找到答案，返回两个元素的下标
        return [i, j];
      }
    }
  }
  //题目保证有解，所以这里理论上不会执行
  return [];
};
```

# 方法 2

```js
var twoSum = function (nums, target) {
  //哈希表，用来存储数组元素及其下标
  const map = new Map();

  for (let i = 0; i < nums.length; i++) {
    // 计算补数
    const complement = target - nums[i];

    //如果补数已经在 map 中，说明找到了两个数相加等于 target
    if (map.has(complement)) {
      // 返回【补数的下标】和【当前数的下标】
      return [map.get(complement), i];
    }

    //如果没找到补数，当前元素和它的下标存入哈希表
    map.set(nums[i], i);
  }
};
```
